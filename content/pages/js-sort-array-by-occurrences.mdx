category: Snippets
title: Sort Array by Occurrences (JS/TS)
description: How to sort the items in an array by the number of their occurrences
---

Ascending (from the **least frequent** to the **most frequent**):

```ts
const items = ['apple', 'pear', 'apple', 'orange', 'apple', 'pear']
// 3x apple, 2x pear, 1x orange

const allItems: Record<string, number> = {}

items.forEach((item) => {
  allItems[item] = allItems[item] ? allItems[item] + 1 : 1
})

const sortedItems = Object.keys(
  Object.fromEntries(Object.entries(allItems).sort(([, a], [, b]) => a - b))
)

console.log(sortedItems)
// ['orange', 'pear', 'apple']
```

Descending (from the **most frequent** to the **least frequent**):

```ts
const items = ['apples', 'bananas', 'bananas', 'bananas', 'oranges', 'oranges']
// 3x apple, 2x pear, 1x orange

const allItems: Record<string, number> = {}

items.forEach((item) => {
  allItems[item] = allItems[item] ? allItems[item] + 1 : 1
})

const sortedItems = Object.keys(
  Object.fromEntries(Object.entries(allItems).sort(([, a], [, b]) => b - a))
)

console.log(sortedItems)
// ['apple', 'pear', 'orange']
```
